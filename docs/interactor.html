 234    def add_function(
 235        self,
 236        external_callable: Callable,
 237        name: Optional[str] = None,
 238        description: Optional[str] = None,
 239        override: bool = False,
 240        disabled: bool = False
 241    ):
 242        """
 243        Register a function for LLM tool calling with full type hints and metadata.
 244
 245        Args:
 246            external_callable (Callable): The function to register.
 247            name (Optional[str]): Optional custom name. Defaults to function's __name__.
 248            description (Optional[str]): Optional custom description. Defaults to first line of docstring.
 249            override (bool): If True, replaces an existing tool with the same name.
 250            disabled (bool): If True, registers the function in a disabled state.
 251
 252        Raises:
 253            ValueError: If the callable is invalid or duplicate name found without override.
 254
 255        Example:
 256            interactor.add_function(my_tool, override=True, disabled=False)
 257        """
 258        if not self.tools_enabled:
 259            return
 260        if not external_callable:
 261            raise ValueError("A valid external callable must be provided.")
 262
 263        function_name = name or external_callable.__name__
 264        docstring = inspect.getdoc(external_callable)
 265        description = description or (docstring.split("\n")[0] if docstring else "No description provided.")
 266
 267        # Remove existing entry if override is enabled
 268        if override:
 269            self.delete_function(function_name)
 270        elif any(t["function"]["name"] == function_name for t in self.tools):
 271            raise ValueError(f"Function '{function_name}' is already registered. Use override=True to replace.")
 272
 273        signature = inspect.signature(external_callable)
 274        properties = {}
 275        required = []
 276
 277        for param_name, param in signature.parameters.items():
 278            param_type = param.annotation
 279            if param_type in (float, int):
 280                type_str = "number"
 281            elif param_type in (str, inspect.Parameter.empty):
 282                type_str = "string"
 283            elif param_type == bool:
 284                type_str = "boolean"
 285            elif param_type == list:
 286                type_str = "array"
 287            else:
 288                type_str = "object"
 289
 290            properties[param_name] = {
 291                "type": type_str,
 292                "description": f"{param_name} parameter"
 293            }
 294
 295            if param.default == inspect.Parameter.empty:
 296                required.append(param_name)
 297
 298        tool_spec = {
 299            "type": "function",
 300            "function": {
 301                "name": function_name,
 302                "description": description,
 303                "parameters": {
 304                    "type": "object",
 305                    "properties": properties,
 306                    "required": required
 307                }
 308            }
 309        }
 310
 311        if disabled:
 312            tool_spec["function"]["disabled"] = True
 313
 314        self.tools.append(tool_spec)
 315        setattr(self, function_name, external_callable)

 318    def disable_function(self, name: str) -> bool:
 319        """
 320        Disable a registered tool function by name.
 321
 322        This marks the function as inactive for tool calling without removing it from the internal registry.
 323        The function remains visible in the tool listing but is skipped during tool selection by the LLM.
 324
 325        Args:
 326            name (str): The name of the function to disable.
 327
 328        Returns:
 329            bool: True if the function was found and disabled, False otherwise.
 330

 333        """
 334        for tool in self.tools:
 335            if tool["function"]["name"] == name:
 336                tool["function"]["disabled"] = True
 337                return True
 338        return False

 363    def delete_function(self, name: str) -> bool:
 364        """
 365        Permanently remove a registered tool function from the Interactor.
 366
 367        This deletes both the tool metadata and the callable attribute, making it fully inaccessible
 368        from the active session. Useful for dynamically trimming the toolset.
 369
 370        Args:
 371            name (str): The name of the function to delete.
 372
 373        Returns:
 374            bool: True if the function was found and removed, False otherwise.
 375
 376        Example:
 377            interactor.delete_function("extract_text")
 378        """
 379        before = len(self.tools)
 380        self.tools = [tool for tool in self.tools if tool["function"]["name"] != name]
 381        if hasattr(self, name):
 382            delattr(self, name)
 383        return len(self.tools) < before

 410

 431                return []

 446                pass

 450    async def _retry_with_backoff(self, func: Callable, *args, **kwargs):
 451        """Execute a function with retry logic and exponential backoff.
 452
 453        Args:
 454            func: The function to execute.
 455            *args: Positional arguments for the function.
 456            **kwargs: Keyword arguments for the function.
 457
 458        Returns:
 459            The result of the function call.
 460
 461        Raises:
 462            Exception: If all retries fail.
 463        """
 464        for attempt in range(self.max_retries + 1):
 465            try:
 466                return await func(*args, **kwargs)
 467            except (RateLimitError, APIConnectionError, aiohttp.ClientError) as e:
 468                if attempt == self.max_retries:
 469                    logger.error(f"All {self.max_retries} retries failed: {e}")
 470                    raise
 471                delay = self.retry_delay * (2 ** attempt)
 472                logger.warning(f"Retry {attempt + 1}/{self.max_retries} after {delay}s due to {e}")
 473                self._log(f"[RETRY] Attempt {attempt + 1}/{self.max_retries} failed: {e}", level="warning")
 474                if attempt == self.max_retries:
 475                    self._log(f"[RETRY] All {self.max_retries} attempts failed. Error: {e}", level="error")
 476                    raise
 477                await asyncio.sleep(delay)
 478            except OpenAIError as e:
 479                logger.error(f"OpenAI error: {e}")
 480                raise
 481            except Exception as e:
 482                logger.error(f"Unexpected error: {e}")
 483                raise

 522    async def interact_async(
 523        self,
 524        user_input: Optional[str],
 525        quiet: bool = False,
 526        tools: bool = True,
 527        stream: bool = True,
 528        markdown: bool = False,
 529        model: Optional[str] = None,
 530        output_callback: Optional[Callable[[str], None]] = None,
 531        session_id: Optional[str] = None
 532    ) -> Optional[str]:
 533        """Asynchronously interact with the AI, handling streaming and tool calls.
 534
 535        Args:
 536            user_input: The user's input message to send to the AI.
 537            quiet: If True, suppresses console output.
 538            tools: Enable (True) or disable (False) tool calling for this interaction.
 539            stream: Enable (True) or disable (False) streaming responses for this interaction.
 540            markdown: If True, renders responses as markdown in the console.
 541            model: Optional model to use for this interaction, overriding the current model.
 542            output_callback: Optional callback to handle each token output.
 543            session_id: Optional session ID for session tracking.
 544
 545        Returns:
 546            str: The AI's response, or None if user_input is empty.
 547        """
 548        if not user_input:
 549            logger.warning("Empty user input provided")
 550            return None
 551
 552        self._log(f"[USER] {user_input}")
 553
 554        if self.encoding:
 555            input_tokens = len(self.encoding.encode(user_input))
 556            if input_tokens > self.context_length:
 557                logger.error(f"User input exceeds max context length: {self.context_length}")
 558                print(f"[red]User Input exceeds max context length:[/red] {self.context_length}")
 559                return None
 560
 561        if model:
 562            provider, model_name = model.split(":", 1)
 563            if provider != self.provider or model_name != self.model:
 564                self._setup_client(model)
 565                self._setup_encoding()
 566            self.provider = provider
 567            self.model = model_name
 568
 569        self.tools_enabled = tools and self.tools_supported
 570
 571        # Session switching logic
 572        if self.session_enabled:
 573            if session_id != self.session_id:
 574                self.session_id = session_id
 575                self._last_session_id = session_id
 576                self.history = self.session.load(session_id) if session_id else []
 577
 578        # Append user message
 579        user_msg = {"role": "user", "content": user_input}
 580        self.history.append(user_msg)
 581        if self.session_enabled and self.session_id:
 582            self.session.msg_insert(self.session_id, user_msg)
 583
 584        if user_input:
 585            self._log(f"[USER] {user_input}")
 586
 587        if self._cycle_messages():
 588            logger.error("Context length exceeded after cycling messages")
 589            return None
 590
 591        use_stream = self.stream if stream is None else stream
 592        content = ""
 593        live = Live(console=console, refresh_per_second=100) if use_stream and markdown and not quiet else None
 594
 595        while True:
 596            params = {
 597                "model": self.model,
 598                "messages": self.history,
 599                "stream": use_stream
 600            }
 601            if self.tools_supported and self.tools_enabled:
 602                enabled_tools = [tool for tool in self.tools if not tool["function"].get("disabled", False)]
 603                params["tools"] = enabled_tools
 604                params["tool_choice"] = "auto"
 605
 606            try:
 607                response = await self._retry_with_backoff(
 608                    self.async_client.chat.completions.create,
 609                    **params
 610                )
 611                tool_calls = []
 612
 613                if use_stream:
 614                    if live:
 615                        live.start()
 616                    tool_calls_dict = {}
 617                    async for chunk in response:
 618                        delta = chunk.choices[0].delta
 619                        finish_reason = chunk.choices[0].finish_reason
 620
 621                        if delta.content:
 622                            content += delta.content
 623                            if output_callback:
 624                                output_callback(delta.content)
 625                            elif live:
 626                                live.update(Markdown(content))
 627                            elif not markdown and not quiet:
 628                                print(delta.content, end="")
 629
 630                        if delta.tool_calls:
 631                            for tool_call_delta in delta.tool_calls:
 632                                index = tool_call_delta.index
 633                                if index not in tool_calls_dict:
 634                                    tool_calls_dict[index] = {"id": None, "function": {"name": "", "arguments": ""}}
 635                                if tool_call_delta.id:
 636                                    tool_calls_dict[index]["id"] = tool_call_delta.id
 637                                if tool_call_delta.function.name:
 638                                    tool_calls_dict[index]["function"]["name"] = tool_call_delta.function.name
 639                                if tool_call_delta.function.arguments:
 640                                    tool_calls_dict[index]["function"]["arguments"] += tool_call_delta.function.arguments
 641                                    if tool_calls_dict[index]["function"]["name"] in self.reveal_tool:
 642                                        if output_callback:
 643                                            output_callback(tool_call_delta.function.arguments)
 644                                        elif live:
 645                                            live.update(Markdown(tool_call_delta.function.arguments))
 646                                        elif not markdown and not quiet:
 647                                            print(tool_call_delta.function.arguments, end="")
 648
 649                    tool_calls = list(tool_calls_dict.values())
 650                    if live:
 651                        live.stop()
 652
 653                else:
 654                    message = response.choices[0].message
 655                    tool_calls = message.tool_calls or []
 656                    if not tool_calls:
 657                        content += message.content or "No response."
 658                        if output_callback:
 659                            output_callback(message.content or "No response.")
 660                        elif not quiet:
 661                            self._render_content(content, markdown, live=None)
 662                        break
 663
 664                if not tool_calls:
 665                    break
 666
 667                assistant_msg = {
 668                    "role": "assistant",
 669                    "content": None,
 670                    "tool_calls": [{
 671                        "id": call["id"] if isinstance(call, dict) else call.id,
 672                        "type": "function",
 673                        "function": {
 674                            "name": call["function"]["name"] if isinstance(call, dict) else call.function.name,
 675                            "arguments": call["function"]["arguments"] if isinstance(call, dict) else call.function.arguments
 676                        }
 677                    } for call in tool_calls]
